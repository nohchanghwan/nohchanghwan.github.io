# 컴포넌트에서 State에 접근하는 4가지 방법

스토어의 state를 컴포넌트에서 접근하는 방법이 여러가지가 있는데

# 1. 템플릿에 this.$store.state.속성 바인딩

<div v-for="item in this.$store.state.속성" v-bind:key="item">{{ item.title }}</div>

# 2. computed 속성에 this.$store.state.속성

const Counter = {
  template: `<div>{{ count }}</div>`,
  computed: {
    count () {
      return this.$store.state.count
    }
  }
}

# 3. mapState

import { mapState } from 'vuex'
  template: <li v-for="item in mappingAsk" v-bind:key="item" class="post">
  computed: {
    //  mapState로 상태 불러오기
    ...mapState({
      mappingAsk: state => state.속성 
    })
